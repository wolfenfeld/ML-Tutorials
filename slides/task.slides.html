def preprocess_data(raw_data):

    preprocessed_data = raw_data.copy()

    # Converting to time objects
    preprocessed_data['reminder_time'] = pd.to_datetime(
        preprocessed_data.reminder_time)
    preprocessed_data['user_created'] = pd.to_datetime(
        preprocessed_data.user_created)

    preprocessed_data['day_of_year'] = preprocessed_data.reminder_time.dt.dayofyear
    preprocessed_data['age'] = 2018 - preprocessed_data.birthyear.astype(int)
    preprocessed_data['days_subscribed'] = (pd.Timestamp.today()-preprocessed_data.user_created).dt.days

    # One hot encoding for categorical data
    preprocessed_data = preprocessed_data.join(
        pd.get_dummies(preprocessed_data['reminder_status']))

    preprocessed_data = preprocessed_data.join(
        pd.get_dummies(preprocessed_data['gender']))rename(
        columns={'U':'no_gender','F':'female','M':'male'})

    preprocessed_data = preprocessed_data.join(
        pd.get_dummies(preprocessed_data['os']))

    return preprocessed_data

def compute_user_mean_data(preprocessed_data):

    aggrigation_dict = {'taken':'mean',
                        'pending':'mean',
                        'dismissed':'mean',
                        'snooze':'mean',
                        'missed':'mean',
                        'snooze_counter':'mean'}

    user_mean_daily_data = preprocessed_data.groupby(
        ['user_id','day_of_year'],
        as_index=False).agg(aggrigation_dict)

    user_mean_data = user_mean_daily_data.groupby(
        ['user_id'], as_index=False).agg(aggrigation_dict)

    user_mean_data = pd.merge(
        left=user_mean_data,
        right=preprocessed_data.groupby(
            'user_id',as_index=False).agg({'med_id':'nunique'}),
        on='user_id').rename(columns={'med_id':'num_of_meds'})

    user_mean_data = pd.merge(
        left=user_mean_data,
        right=preprocessed_data.groupby(
            ['user_id','day_of_year'],as_index=False).agg(
            {'med_id':'count'}).groupby(
            ['user_id'],as_index=False).agg({'med_id':'mean'}),
        on='user_id').rename(columns={'med_id':'mean_daily_doses'})

    return user_mean_data

def compute_rolling_mean_adherence(preprocessed_data):

    preprocessed_data['rolling_mean'] = preprocessed_data.groupby(
        ['user_id'],as_index=False)['taken'].rolling(
        7,min_periods=1).mean().reset_index()['taken']

    rolling_mean_adherence = pd.pivot_table(
        preprocessed_data,
        values=['rolling_mean'],
        index=['user_id'],
        columns=['day_of_year']).fillna(0).reset_index()

    rolling_mean_adherence.columns = [
        'rolling_mean_{0}'.format(col_name) for
        col_name in rolling_mean_adherence.columns.get_level_values(1)]

    return rolling_mean_adherence.rename(columns={'rolling_mean_':'user_id'})

def compute_labels(preprocessed_data):

    data_labels = preprocessed_data.groupby(
        ['user_id','day_of_year'],as_index=False).agg(
        {'taken':'mean'}).groupby(
        ['user_id'],as_index=False).agg(
        {'taken':'mean'}).rename(columns={'taken':'adherence'})

    data_labels['adherent_user'] = data_labels['adherence'] > 0.8

    return data_labels

def process_data(preprocessed_data, prediction_start_day=225):

    labels = compute_labels(preprocessed_data[
        preprocessed_data.day_of_year >= prediction_start_day].copy())

    preprocessed_data = preprocessed_data[
        preprocessed_data.day_of_year < prediction_start_day].copy()

    processed_data = compute_user_mean_data(preprocessed_data)

    processed_data = pd.merge(left=processed_data,
                              right=labels,
                              on='user_id')

    user_data = compute_user_data(preprocessed_data)

    processed_data = pd.merge(left=processed_data,
                              right=user_data,
                              on='user_id')

    rolling_mean_adherence = compute_rolling_mean_adherence(preprocessed_data)

    processed_data = pd.merge(left=processed_data,
                              right=rolling_mean_adherence,
                              on='user_id')

    return processed_data
